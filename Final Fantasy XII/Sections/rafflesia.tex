\chapter{Rafflesia}

\begin{enumerate}
	\item \pickup{5040 Gil}{Balfonheim}
	\item Teleport to Mr. Bur-Omisace
	\item If you weren't able to buy the Glacial Staves before, then by them now
	\item Take the Chocobo to the Feywood. Dismount before the First Zone Line in the Feywood
	\item After the Zone Line:
		\begin{itemize}
			\penelof \fira\ \ashe, stop running
			\ashef Cure Reserve, start running
			\ashef Cure \vaan, so that we can do too-far-away despawning
			\item Despawn enemies along the entire way by switching the leader to \ashe\ and then back to \vaan
			\vaanf Remove Mirror Mail
		\end{itemize}
	\item Once you get to the Save Crystal Zone:
	\begin{itemize}
		\decoy{\ashe}{\vaan}
		\item \castprotect{\vaan}{\vaan}
		\item \GirlsGambitOn, so that they both start to cast Reflect at approximately the same time.
		\item Once the Reflects land:
		\begin{equip}
			\begin{itemize}
			\optimize{\vaan}
			\penelof Cherry Staff
			\ashef Cherry Staff
			\end{itemize}
		\end{equip}
		\begin{menu}
			\party{\vaan}
		\end{menu}
	\end{itemize}
\end{enumerate}
